package mathNode;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

/**
 * Unit tests for {@link Operator#getRightNode()}.
 */
class OperatorTest {

    // Minimal stub for Expression to allow compilation
    static class Expression {}

    // Minimal stub for Operator to allow compilation and control of rightNode field
    static class Operator {
        private final Expression rightNode;

        public Operator(Expression rightNode) {
            this.rightNode = rightNode;
        }

        public Expression getRightNode() {
            return rightNode;
        }
    }

    @Test
    void test_getRightNode_whenRightNodeIsNull_returnsNull() {
        // Arrange
        Operator operator = new Operator(null);

        // Act
        Expression result = operator.getRightNode();

        // Assert
        assertNull(result);
    }

    @Test
    void test_getRightNode_whenRightNodeIsNonNull_returnsSameInstance() {
        // Arrange
        Expression expected = new Expression();
        Operator operator = new Operator(expected);

        // Act
        Expression result = operator.getRightNode();

        // Assert
        assertSame(expected, result);
    }
}
