        Int intNode = new Int();

        // Assert
        assertTrue(result, "checkTree should always return true");

    @ParameterizedTest

        // Act
        boolean result = intNode.checkTree();

        assertTrue(result, "checkTree should return true regardless of internal state or value: " + value);
    }
